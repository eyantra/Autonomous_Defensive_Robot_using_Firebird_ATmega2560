00030                                                 PORTD 3 --> TXD1 UART1 transmit for RS232 serial communication
00031 
00032                                                 PORTH 0 --> RXD2 UART 2 receive for USB - RS232 communication
00033                                                 PORTH 1 --> TXD2 UART 2 transmit for USB - RS232 communication

00035                                                 PORTE 0 --> RXD0 UART0 receive for ZigBee wireless communication
00036                                                 PORTE 1 --> TXD0 UART0 transmit for ZigBee wireless communication

00039                                                 PORTJ 1 --> TXD3 UART3 transmit available on microcontroller expainsion board

00208             //changing the degree value to shoot in a range of 10 degrees(degree-8,degree+2) continuously(in steps of 2 degrees).
00209             degree = (degree >= 8) ? (degree - 8) : 0;
00210             int p = 0;
00211             for (p = 0; p < 10 && degree < 360; p++) {
00212                 if (degree <= 180) {    //degree value < 180 thus, only one servo motor needs to move(the bottom one).
00213                     servo_1(degree);
00214                     servo_2(180);
00215                 } else {                //degree valu > 180 thus, both motors need to move.
00216                     servo_1(180);

00218                 }
00219                 degree += 2;

00221             }
00222         }

00224     }
